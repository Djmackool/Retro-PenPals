function normalizeList(text) {
  return text.split(",").map(x => x.trim().toLowerCase()).filter(Boolean);
}

function getProfile() {
  return {
    name: document.getElementById("name").value.trim(),
    state: document.getElementById("state").value.trim(),
    city: document.getElementById("city").value.trim(),
    mailType: document.getElementById("mailType").value,
    frequency: document.getElementById("frequency").value,
    interests: document.getElementById("interests").value.trim(),
    dislikes: document.getElementById("dislikes").value.trim(),
    bio: document.getElementById("bio").value.trim()
  };
}

function saveProfile(showAlert = true) {
  localStorage.setItem("snailmail-codepen-profile", JSON.stringify(getProfile()));
  if (showAlert) alert("Profile saved on this device.");
}

function loadSaved() {
  const raw = localStorage.getItem("snailmail-codepen-profile");
  if (!raw) return;
  const profile = JSON.parse(raw);
  Object.entries(profile).forEach(([k, v]) => {
    const el = document.getElementById(k);
    if (el) el.value = v;
  });
}

function resetAll() {
  localStorage.removeItem("snailmail-codepen-profile");
  ["name","state","city","mailType","frequency","interests","dislikes","bio","interestSearch","stateSearch"].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.value = "";
  });
  renderResults(sampleProfiles.map(p => ({...p, score: 80})));
}

function scoreMatch(profile, person, interestQuery, stateQuery) {
  let score = 70;
  const myInterests = normalizeList(profile.interests || "");
  const myDislikes = normalizeList(profile.dislikes || "");
  const theirInterests = person.interests.map(x => x.toLowerCase());
  const theirDislikes = person.dislikes.map(x => x.toLowerCase());

  myInterests.forEach(item => {
    if (theirInterests.some(t => t.includes(item) || item.includes(t))) score += 6;
  });

  myDislikes.forEach(item => {
    if (theirInterests.some(t => t.includes(item) || item.includes(t))) score -= 8;
  });

  theirDislikes.forEach(item => {
    if (myInterests.some(t => t.includes(item) || item.includes(t))) score -= 6;
  });

  if (profile.state && profile.state.toLowerCase() === person.state.toLowerCase()) score += 8;
  if (profile.mailType && profile.mailType === person.mailType) score += 5;
  if (profile.frequency && profile.frequency === person.frequency) score += 5;

  if (interestQuery) {
    const q = interestQuery.toLowerCase();
    if (theirInterests.some(t => t.includes(q) || q.includes(t))) score += 12;
    else score -= 8;
  }

  if (stateQuery) {
    if (person.state.toLowerCase().includes(stateQuery.toLowerCase())) score += 12;
    else score -= 10;
  }

  return Math.max(1, Math.min(99, score));
}

function runSearch() {
  saveProfile(false);

  const profile = JSON.parse(localStorage.getItem("snailmail-codepen-profile") || "{}");
  const interestQuery = document.getElementById("interestSearch").value.trim();
  const stateQuery = document.getElementById("stateSearch").value.trim();

  let filtered = sampleProfiles.filter(person => {
    const matchesState = !stateQuery || person.state.toLowerCase().includes(stateQuery.toLowerCase());
    const matchesInterest = !interestQuery || person.interests.some(i => i.toLowerCase().includes(interestQuery.toLowerCase()));
    return matchesState && matchesInterest;
  });

  filtered = filtered.map(person => ({
    ...person,
    score: scoreMatch(profile, person, interestQuery, stateQuery)
  })).sort((a,b) => b.score - a.score);

  renderResults(filtered);
}

function renderResults(matches) {
  const container = document.getElementById("results");
  if (!matches.length) {
    container.innerHTML = '<div class="empty">No profiles match that search yet. Try a broader interest like <strong>journaling</strong>, <strong>tattoos</strong>, or <strong>postcards</strong>.</div>';
    return;
  }

  container.innerHTML = matches.map(person => `
    <div class="match">
      <div>
        <h3>${person.name}</h3>
        <div>${person.city}, ${person.state}</div>
        <div class="meta">
          <span class="tag">${person.mailType}</span>
          <span class="tag">${person.frequency}</span>
        </div>

        <div class="section-title">Interests</div>
        <div class="tags">${person.interests.map(i => `<span class="tag">${i}</span>`).join("")}</div>

        <div class="section-title">Dislikes</div>
        <div class="tags">${person.dislikes.map(i => `<span class="tag">${i}</span>`).join("")}</div>

        <div class="section-title">About</div>
        <div>${person.bio}</div>
      </div>
      <div class="score">${person.score}%<br><span style="font-size:12px;font-weight:400">match</span></div>
    </div>
  `).join("");
}
